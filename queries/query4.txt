PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX sdm: <http://www.bdma.sdm#>
SELECT ?title WHERE {
    ?paper rdf:type sdm:Paper .
    ?proceedings rdf:type sdm:Proceedings .
    ?conference rdf:type sdm:Conference .
    ?researcharea rdf:type sdm:ResearchArea .
    ?author rdf:type sdm:Author .
    ?paper sdm:publishedin ?proceedings ;
          	sdm:paper_title ?title .
    ?conference sdm:hasproceedings ?proceedings .
    ?conference sdm:venuerelatedto ?researcharea .
    ?researcharea sdm:topic "Big data" .
    ?author sdm:wrotepaper ?paper ;
            sdm:fullName "Albert Bifet" .
}
